The Transformation: Live Demo

query projectDetails($id: Int!) {
  projectDetails(id: $id) {
    users(active: true, excludeEmployees: true) {
      profile { firstName, lastName, email }
      offers { id, jobTitle, status }
    }
    departments { name, memberCount }
    customFields { name, required }
  }
}
